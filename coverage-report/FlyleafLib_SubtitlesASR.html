 547                {

 552                {

 555                }

 561            // For this purpose, the ffmpeg API is used to perform the conversion.
 562            // Audio files are divided by a certain size, stored in memory, and passed by memory stream.

 564            int targetChannel = 1;

 566            TimeSpan waveDuration = TimeSpan.Zero; // for logging
 567
 568            const int waveHeaderSize = 44;
 569
 570            // Stream processing is performed by dividing the audio by a certain size and passing it to whisper.
 571            long chunkSize = _config.Subtitles.ASRChunkSize;
 572            // Also split by elapsed seconds for live
 573            TimeSpan chunkElapsed = TimeSpan.FromSeconds(_config.Subtitles.ASRChunkSeconds);
 574            Stopwatch chunkSw = new();

 579            int chunkCnt = 0;
 580            TimeSpan? chunkStart = null;
 581            long framePts = AV_NOPTS_VALUE;
 582
 583            int demuxErrors = 0;
 584            int decodeErrors = 0;
 585
 586            while (!token.IsCancellationRequested)
 587            {
 588                _demuxer.Interrupter.ReadRequest();

 590
 591                if (ret != 0)
 592                {
 593                    av_packet_unref(_packet);
 594
 595                    if (_demuxer.Interrupter.Timedout)
 596                    {
 597                        if (token.IsCancellationRequested)
 598                            break;

 600                        ret.ThrowExceptionIfError("av_read_frame (timed out)");
 601                    }
 602
 603                    if (ret == AVERROR_EOF || token.IsCancellationRequested)
 604                    {
 605                        break;
 606                    }
 607
 608                    // demux error
 609                    if (CanWarn) Log.Warn($"av_read_frame: {FFmpegEngine.ErrorCodeToMsg(ret)} ({ret})");
 610
 611                    if (++demuxErrors == _config.Demuxer.MaxErrors)
 612                    {
 613                        ret.ThrowExceptionIfError("av_read_frame");
 614                    }
 615                    continue;
 616                }

 619                if (_packet->stream_index != _stream.StreamIndex)
 620                {
 621                    av_packet_unref(_packet);
 622                    continue;
 623                }

 630                    if (ret == AVERROR(EAGAIN))

 633                        ret.ThrowExceptionIfError("avcodec_send_packet (EAGAIN)");

 636                    // decoder error
 637                    if (CanWarn) Log.Warn($"avcodec_send_packet: {FFmpegEngine.ErrorCodeToMsg(ret)} ({ret})");

 639                    if (++decodeErrors == _config.Decoder.MaxErrors)
 640                    {
 641                        ret.ThrowExceptionIfError("avcodec_send_packet");
 642                    }

 644                    continue;
 645                }

 651                    {
 652                        break;
 653                    }
 654                    ret.ThrowExceptionIfError("avcodec_receive_frame");

 661                    {
 662                        framePts = _frame->pts;
 663                    }

 667                        framePts += _frame->duration;
 668                    }
 669
 670                    waveDuration = waveDuration.Add(new TimeSpan((long)(_frame->duration * _stream.Timebase)));
 671
 672                    if (chunkStart == null)
 673                    {
 674                        chunkStart = new TimeSpan((long)(framePts * _stream.Timebase) - _demuxer.StartTime);

 676                        {
 677                            // Correct to 0 if negative
 678                            chunkStart = new TimeSpan(0);
 679                        }
 680                    }
 681
 682                    ResampleTo(waveStream, _frame, targetSampleRate, targetChannel);
 683
 684                    // TODO: L: want it to split at the silent part
 685                    if (waveStream.Length >= chunkSize || chunkSw.Elapsed >= chunkElapsed)
 686                    {
 687                        TimeSpan chunkEnd = new TimeSpan((long)(framePts * _stream.Timebase) - _demuxer.StartTime);

 690                        if (CanInfo) Log.Info(

 697                        if (CanDebug) Log.Debug($"Writing chunk to channel ({chunkCnt})");

 699                        channel.Writer.WriteAsync(chunk, token).AsTask().Wait(token);

 703                            waveStream = stream;
 704                        else
 705                            waveStream = new MemoryStream();

 707                        WriteWavHeader(waveStream, targetSampleRate, targetChannel);
 708                        waveDuration = TimeSpan.Zero;

 710                        chunkStart = null;
 711                        chunkSw.Restart();
 712                        framePts = AV_NOPTS_VALUE;
 713                    }
 714                }
 715            }

 717            token.ThrowIfCancellationRequested();

 719            // Process remaining

 722                TimeSpan chunkEnd = new TimeSpan((long)(framePts * _stream.Timebase) - _demuxer.StartTime);

 724                chunkCnt++;

 732
 733                if (CanDebug) Log.Debug($"Writing last chunk to channel ({chunkCnt})");
 734                channel.Writer.WriteAsync(chunk, token).AsTask().Wait(token);
 735                if (CanDebug) Log.Debug($"Done writing last chunk to channel ({chunkCnt})");
 736            }
 737        }

 740    private static void WriteWavHeader(Stream stream, int sampleRate, int channels)
 741    {
 742        using BinaryWriter writer = new(stream, Encoding.UTF8, true);
 743        writer.Write(['R', 'I', 'F', 'F']);
 744        writer.Write(0); // placeholder for file size
 745        writer.Write(['W', 'A', 'V', 'E']);
 746        writer.Write(['f', 'm', 't', ' ']);
 747        writer.Write(16); // PCM header size
 748        writer.Write((short)1); // PCM format
 749        writer.Write((short)channels);
 750        writer.Write(sampleRate);
 751        writer.Write(sampleRate * channels * 2); // Byte rate
 752        writer.Write((short)(channels * 2)); // Block align
 753        writer.Write((short)16); // Bits per sample
 754        writer.Write(['d', 'a', 't', 'a']);
 755        writer.Write(0); // placeholder for data size
 756    }

 758    private static void UpdateWavHeader(Stream stream)
 759    {
 760        long fileSize = stream.Length;
 761        stream.Seek(4, SeekOrigin.Begin);
 762        stream.Write(BitConverter.GetBytes((int)(fileSize - 8)), 0, 4);
 763        stream.Seek(40, SeekOrigin.Begin);
 764        stream.Write(BitConverter.GetBytes((int)(fileSize - 44)), 0, 4);
 765        stream.Position = 0;
 766    }

 768    private byte[] _sampledBuf = [];
 769    private int _sampledBufSize;
 770
 771    // for codec change detection
 772    private int _lastFormat;
 773    private int _lastSampleRate;
 774    private ulong _lastChannelLayout;
 775
 776    private unsafe void ResampleTo(Stream toStream, AVFrame* frame, int targetSampleRate, int targetChannel)
 777    {
 778        bool codecChanged = false;
 779
 780        if (_lastFormat != frame->format)
 781        {
 782            _lastFormat = frame->format;
 783            codecChanged = true;
 784        }
 785        if (_lastSampleRate != frame->sample_rate)
 786        {
 787            _lastSampleRate = frame->sample_rate;
 788            codecChanged = true;
 789        }
 790        if (_lastChannelLayout != frame->ch_layout.u.mask)
 791        {
 792            _lastChannelLayout = frame->ch_layout.u.mask;
 793            codecChanged = true;
 794        }
 795
 796        // Reinitialize SwrContext because codec changed
 797        // Note that native error will occur if not reinitialized.
 798        // Reference: AudioDecoder::RunInternal
 799        if (_swrContext != null && codecChanged)
 800        {
 801            fixed (SwrContext** ptr = &_swrContext)
 802            {
 803                swr_free(ptr);
 804            }
 805            _swrContext = null;
 806        }
 807
 808        if (_swrContext == null)
 809        {
 810            AVChannelLayout outLayout;
 811            av_channel_layout_default(&outLayout, targetChannel);
 812
 813            // NOTE: important to reuse this context
 814            fixed (SwrContext** ptr = &_swrContext)
 815            {
 816                swr_alloc_set_opts2(
 817                    ptr,
 818                    &outLayout,
 819                    AVSampleFormat.S16,
 820                    targetSampleRate,
 821                    &frame->ch_layout,
 822                    (AVSampleFormat)frame->format,
 823                    frame->sample_rate,
 824                    0, null)
 825                    .ThrowExceptionIfError("swr_alloc_set_opts2");
 826
 827                swr_init(_swrContext)
 828                    .ThrowExceptionIfError("swr_init");
 829            }
 830        }
 831
 832        // ffmpeg ref: https://github.com/FFmpeg/FFmpeg/blob/504df09c34607967e4109b7b114ee084cf15a3ae/libavfilter/af_are
 833        double ratio = targetSampleRate * 1.0 / frame->sample_rate; // 16000:44100=0.36281179138321995
 834        int nOut = (int)(frame->nb_samples * ratio) + 32;
 835
 836        long delay = swr_get_delay(_swrContext, targetSampleRate);
 837        if (delay > 0)
 838        {
 839            nOut += (int)Math.Min(delay, Math.Max(4096, nOut));
 840        }
 841        int needed = nOut * targetChannel * sizeof(ushort);
 842
 843        if (_sampledBufSize < needed)
 844        {
 845            _sampledBuf = new byte[needed];
 846            _sampledBufSize = needed;
 847        }
 848
 849        int samplesPerChannel;
 850
 851        fixed (byte* dst = _sampledBuf)
 852        {
 853            samplesPerChannel = swr_convert(
 854                 _swrContext,
 855                 &dst,
 856                 nOut,
 857                 frame->extended_data,
 858                 frame->nb_samples);
 859        }
 860        samplesPerChannel.ThrowExceptionIfError("swr_convert");
 861
 862        int resampledDataSize = samplesPerChannel * targetChannel * sizeof(ushort);
 863
 864        toStream.Write(_sampledBuf, 0, resampledDataSize);
 865    }
